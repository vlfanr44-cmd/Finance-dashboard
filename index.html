const SHEET_NAME = 'data'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const response = ContentService.createTextOutput(); response.setMimeType(ContentService.MimeType.JSON); try { const action = e?.parameter?.action || (e?.postData ? JSON.parse(e.postData.contents).action : null); if (action === 'get') { response.setContent(JSON.stringify(getData())); } else if (action === 'save') { const body = JSON.parse(e.postData.contents); response.setContent(JSON.stringify(saveData(body.key, body.value))); } else if (action === 'clear') { response.setContent(JSON.stringify(clearAll())); } else { response.setContent(JSON.stringify({ ok: false, error: 'unknown action' })); } } catch (err) { response.setContent(JSON.stringify({ ok: false, error: err.message })); } return response; } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['key', 'value', 'updatedAt']); sheet.setFrozenRows(1); } return sheet; } function getData() { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { ok: true, data: null, log: null }; const result = {}; rows.slice(1).forEach(row => { result[row[0]] = row[1] ? JSON.parse(row[1]) : null; }); return { ok: true, data: result['__data__'] || null, log: result['__log__'] || null }; } function saveData(key, value) { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === key) { sheet.getRange(i + 1, 1, 1, 3).setValues([[key, JSON.stringify(value), new Date().toISOString()]]); return { ok: true }; } } sheet.appendRow([key, JSON.stringify(value), new Date().toISOString()]); return { ok: true }; } function clearAll() { const sheet = getSheet(); const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); return { ok: true }; }
